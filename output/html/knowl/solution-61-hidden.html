\begin{gather*} P(A_1) = 0.2 \quad P(A_2) = 0.35 \quad P(A_3) = 0.45\\ P(B \mid A_1) = 0.7 \quad P(B \mid A_2) = 0.25 \quad P(B \mid A_3) = 0.05 \end{gather*}
